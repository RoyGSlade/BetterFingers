# requirements.txt — unpinned, cross-platform runtime manifest.
#
# This mirrors requirements.in and stays intentionally unpinned so a single
# file installs on Linux, Windows, and macOS (the Windows-only deps carry
# environment markers). It remains the install target for the packaging and
# e2e workflows (build-installer.yml, e2e-screenshots.yml).
#
# For reproducible, hash-verified installs, prefer the generated lock for your
# platform instead:
#     Linux    pip install --require-hashes -r requirements-linux.lock
#     Windows  pip install --require-hashes -r requirements-win.lock   (see file)
# Edit requirements.in (the source of truth) and regenerate the locks with
# pip-compile; keep this file in sync with it. Dev/test deps live in
# requirements-dev.in / requirements-dev.lock.

# ---- Core ----
Pillow                      # Image processing (tray icon, overlays)
psutil                      # Hardware metrics (RAM / cores for hardware_report)
numpy                       # Numeric arrays for audio processing
scipy                       # Signal processing (audio gate)

# ---- Audio & Speech ----
sounddevice                 # Microphone capture
faster-whisper              # Whisper speech-to-text (CTranslate2)
kokoro; python_version < "3.13"   # TTS backend (primary)
kokoro-onnx                 # TTS backend (ONNX variant)

# ---- Input & Clipboard ----
keyboard                    # Global hotkey listener
pygame                      # Controller / gamepad support
pydirectinput; sys_platform == "win32"   # Simulated keystrokes (Windows-only; ctypes.windll)
pyperclip                   # Cross-platform clipboard

# ---- LLM Server ----
requests                    # HTTP client (model downloads)
huggingface_hub             # Model cache management (Whisper)
pyyaml                      # Profile / config files

# ---- Sidecar API ----
mcp                         # MCP client SDK (C12 — persona tool integration)
fastapi                     # REST + WebSocket server
uvicorn                     # ASGI server
websockets                  # WebSocket protocol support
python-multipart            # FastAPI file/form uploads

# ---- Windows Audio Control ----
# Windows-only (COM / Core Audio). On Linux, audio ducking uses `pactl`
# (PulseAudio / PipeWire) at runtime instead.
comtypes; sys_platform == "win32"   # COM automation (Windows)
pycaw; sys_platform == "win32"      # Windows audio session control
