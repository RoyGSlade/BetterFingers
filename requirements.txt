# requirements.txt — NON-AUTHORITATIVE compatibility export. Read this first.
#
# This file is a human-readable, unpinned, cross-platform export for people
# skimming dependencies (the Windows-only deps carry environment markers so a
# single file installs on Linux, Windows, and macOS). It is NOT what CI or
# release builds install from and it is not the source of truth for versions.
#
# The authoritative, hash-pinned dependency sets actually consumed by CI and
# builds are:
#     requirements-win.lock         Windows runtime (pip install --require-hashes -r ...)
#     requirements-linux.lock       Linux runtime, GPU/CUDA torch
#     requirements-linux-cpu.lock   Linux runtime, CPU-only torch (GPU-less CI runners)
#     requirements-dev.lock / requirements-dev-win.lock   Dev/test deps, per platform
#
# This mirrors requirements.in (the human-maintained source of truth for
# ranges). Edit requirements.in and regenerate the locks with pip-compile;
# keep this file in sync with it — do not hand-edit versions here expecting
# them to affect what actually gets installed by CI/release tooling.

# ---- Core ----
Pillow                      # Image processing (tray icon, overlays)
psutil                      # Hardware metrics (RAM / cores for hardware_report)
numpy                       # Numeric arrays for audio processing
scipy                       # Signal processing (audio gate)

# ---- Audio & Speech ----
sounddevice                 # Microphone capture
faster-whisper              # Whisper speech-to-text (CTranslate2)
kokoro; python_version < "3.13"   # TTS backend (primary)
kokoro-onnx                 # TTS backend (ONNX variant)

# ---- Input & Clipboard ----
keyboard                    # Global hotkey listener
pygame                      # Controller / gamepad support
pydirectinput; sys_platform == "win32"   # Simulated keystrokes (Windows-only; ctypes.windll)
pyperclip                   # Cross-platform clipboard

# ---- LLM Server ----
requests                    # HTTP client (model downloads)
huggingface_hub             # Model cache management (Whisper)
pyyaml                      # Profile / config files

# ---- Sidecar API ----
mcp                         # MCP client SDK (C12 — persona tool integration)
fastapi                     # REST + WebSocket server
uvicorn                     # ASGI server
websockets                  # WebSocket protocol support
python-multipart            # FastAPI file/form uploads

# ---- Windows Audio Control ----
# Windows-only (COM / Core Audio). On Linux, audio ducking uses `pactl`
# (PulseAudio / PipeWire) at runtime instead.
comtypes; sys_platform == "win32"   # COM automation (Windows)
pycaw; sys_platform == "win32"      # Windows audio session control
